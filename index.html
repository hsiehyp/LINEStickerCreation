/* iframe-safe confirm dialog.
* Browser window.confirm() is silently blocked in many iframe contexts (incl. claude.ai),
* so we render our own modal that returns a Promise<boolean>.
*/
function customConfirm(message, opts = {}) {
return new Promise(resolve => {
const overlay = document.createElement("div");
overlay.className = "modal-overlay";
const titleHtml = opts.title ? `<div class="modal-title">${opts.title}</div>` : "";
overlay.innerHTML = `
<div class="modal-dialog" role="dialog" aria-modal="true">
<div class="modal-icon">${opts.icon || "⚠️"}</div>
${titleHtml}
<div class="modal-message">${message}</div>
<div class="modal-actions">
<button class="btn outline" data-act="cancel">${opts.cancelText || "取消"}</button>
<button class="btn ${opts.danger ? "danger" : "primary"}" data-act="confirm">${opts.confirmText || "確認"}</button>
</div>
</div>`;
document.body.appendChild(overlay);

function cleanup(result) {
overlay.remove();
document.removeEventListener("keydown", onKey);
resolve(result);
}
function onKey(e) {
if (e.key === "Escape") { e.preventDefault(); cleanup(false); }
else if (e.key === "Enter") { e.preventDefault(); cleanup(true); }
}
overlay.addEventListener("click", e => {
if (e.target === overlay) cleanup(false); // click outside dialog area
});
overlay.querySelectorAll("[data-act]").forEach(btn => {
btn.addEventListener("click", () => cleanup(btn.dataset.act === "confirm"));
});
document.addEventListener("keydown", onKey);
// Focus the cancel button by default for safety (avoid accidental Enter destroying data)
setTimeout(() => {
const cancelBtn = overlay.querySelector('[data-act="cancel"]');
if (cancelBtn) cancelBtn.focus();
}, 30);
});
}

function fileToImage(file) {
return new Promise((res, rej) => {
const url = URL.createObjectURL(file);
const img = new Image();
img.onload = () => { URL.revokeObjectURL(url); res(img); };
img.onerror = (e) => { URL.revokeObjectURL(url); rej(e); };
img.src = url;
});
}

function dataURLToImage(dataURL) {
return new Promise((res, rej) => {
const img = new Image();
img.onload = () => res(img);
img.onerror = rej;
img.src = dataURL;
});
}

setTimeout(() => URL.revokeObjectURL(url), 1500);
}

function toggleKeyword(kw) {
const i = PromptState.selectedActions.indexOf(kw);
if (i >= 0) {
PromptState.selectedActions.splice(i, 1);
} else {
if (PromptState.selectedActions.length >= MAX_ACTIONS) {
toast(`最多只能選 ${MAX_ACTIONS} 個動作`, "error");
return;
}
PromptState.selectedActions.push(kw);
}
renderSelectedKeywordsBar();
refreshChipStates();
renderPrompt();
}

function renderSelectedKeywordsBar() {

html = `<span class="kw-selected-empty">尚未選取動作（請從下方分類點選 1–${MAX_ACTIONS} 個）</span>`;

html = items.map((kw, i) => `

<span class="idx">${i + 1}</span>
<span>${kw}</span>
<button type="button" data-kw="${kw}" aria-label="移除">×</button>
</span>`).join("");
}
const counterClass = items.length === MAX_ACTIONS ? "kw-counter full" : "kw-counter";
html += `<span class="${counterClass}">已選 <b>${items.length}</b>/${MAX_ACTIONS}</span>`;
bar.innerHTML = html;
bar.querySelectorAll(".kw-selected-chip button").forEach(btn => {
btn.addEventListener("click", e => {
e.stopPropagation();
const kw = btn.dataset.kw;
const i = PromptState.selectedActions.indexOf(kw);

renderSelectedKeywordsBar();
refreshChipStates();
renderPrompt();
});

});

// Bilinear interpolation read from the 100x100 mask, scaled by scalePct
// Returns alpha in [0, 0.985]
function getMaskAlpha(localX, localY, scaleSize) {
// localX, localY are in scaled mask coordinates (0..scaleSize)
if (localX < 0 || localY < 0 || localX >= scaleSize || localY >= scaleSize) return 0;
const u = (localX / scaleSize) * 99; // 0..99
const v = (localY / scaleSize) * 99;
const x0 = Math.floor(u), y0 = Math.floor(v);
const x1 = Math.min(x0 + 1, 99), y1 = Math.min(y0 + 1, 99);
const fx = u - x0, fy = v - y0;
const M = WMState.maskAlpha;
const a00 = M[y0*100 + x0];
const a10 = M[y0*100 + x1];
const a01 = M[y1*100 + x0];
const a11 = M[y1*100 + x1];
let a = a00*(1-fx)*(1-fy) + a10*fx*(1-fy) + a01*(1-fx)*fy + a11*fx*fy;
a *= WMState.alphaMul;
if (a > 0.985) a = 0.985;
if (a < 0) a = 0;
return a;
}

function processWatermark() {
if (!WMState.origImageData || !WMState.maskReady) return;
const W = WMState.imgW, H = WMState.imgH;
const procCtx = WMState.procCanvas.getContext("2d", { willReadFrequently: true });

// If processing is disabled by user, just mirror the original (no Alpha Unmix).
if (!WMState.enabled) {
procCtx.drawImage(WMState.origCanvas, 0, 0);
AppState.watermark.processedImage = WMState.procCanvas;
updateCopyToSplitBtn();

$("#wmStatusPill").style.background = "rgba(251,191,36,.12)";
$("#wmStatusPill").style.borderColor = "rgba(251,191,36,.3)";
$("#wmStatusPill").style.color = "var(--c-warn)";
return;
}

const src = WMState.origImageData.data;
const out = new Uint8ClampedArray(src);
const scaleSize = 100 * (WMState.scalePct / 100);
const x0 = Math.round(W - WMState.rightOff - scaleSize);
const y0 = Math.round(H - WMState.bottomOff - scaleSize);

const xStart = Math.max(0, x0);
const yStart = Math.max(0, y0);
const xEnd = Math.min(W, x0 + Math.ceil(scaleSize) + 1);
const yEnd = Math.min(H, y0 + Math.ceil(scaleSize) + 1);

for (let y = yStart; y < yEnd; y++) {
for (let x = xStart; x < xEnd; x++) {
const a = getMaskAlpha(x - x0, y - y0, scaleSize);
if (a <= 0) continue;
const inv = 1 - a;
const idx = (y * W + x) * 4;
// bg = (obs - 255*a) / (1 - a), per channel
const r = (src[idx] - 255*a) / inv;
const g = (src[idx+1] - 255*a) / inv;
const b = (src[idx+2] - 255*a) / inv;
out[idx] = clamp(r, 0, 255);
out[idx+1] = clamp(g, 0, 255);
out[idx+2] = clamp(b, 0, 255);
}
}
procCtx.putImageData(new ImageData(out, W, H), 0, 0);
AppState.watermark.processedImage = WMState.procCanvas; // store ref for tab3 copy
updateCopyToSplitBtn();
$("#wmStatusPill").style.display = "flex";
$("#wmStatusPill").innerHTML = '<span class="dot"></span>PROCESSED';
$("#wmStatusPill").style.background = "";
$("#wmStatusPill").style.borderColor = "";
$("#wmStatusPill").style.color = "";
}

// Reusable: returns the source-canvas-coordinate boundaries (in px) for the 5x4 grid given current vLines/hLines.
// boundaries.xs has 6 values: [0, vLine0, vLine1, vLine2, vLine3, W]
// boundaries.ys has 5 values: [0, hLine0, hLine1, hLine2, H]
function getSplitBoundaries(W, H) {
const xs = [0, ...AppState.split.vLines.map(p => p * W), W];
const ys = [0, ...AppState.split.hLines.map(p => p * H), H];
return { xs, ys };
}

function drawSplitPreview() {

cv.height = Math.round(H * scaleDisp);

const y = parseInt($("#splitY").value);

const dx = ((W - dW)/2 + x) * scaleDisp;
const dy = ((H - dH)/2 + y) * scaleDisp;

drawSplitLine(ctx, 0, yy, cv.width, yy, isActive);
});

const { xs, ys } = getSplitBoundaries(cv.width, cv.height);

function drawSplitLine(ctx, x1, y1, x2, y2, active) {

ctx.setLineDash([]);
}

/* ----- Interactive line dragging ----- */
const SplitInteract = {
hover: null, // {kind: "v"|"h", idx} or null
drag: null, // {kind, idx, dragging: bool}
HIT_PX: 8 // pixels around line to grab
};

function setupSplitCanvasInteraction() {
const cv = $("#splitCanvas");
if (!cv || cv._splitInteractInited) return;
cv._splitInteractInited = true;

function getCanvasCoord(e) {
const rect = cv.getBoundingClientRect();
const cx = e.clientX - rect.left;
const cy = e.clientY - rect.top;
const sx = cx * (cv.width / rect.width);
const sy = cy * (cv.height / rect.height);
return { x: sx, y: sy };
}

function findHit(x, y, hitPx) {
let best = null, bestDist = hitPx + 1;
AppState.split.vLines.forEach((p, i) => {
const lineX = p * cv.width;
const d = Math.abs(x - lineX);
if (d < bestDist) { bestDist = d; best = { kind: "v", idx: i }; }
});
AppState.split.hLines.forEach((p, i) => {
const lineY = p * cv.height;
const d = Math.abs(y - lineY);
if (d < bestDist) { bestDist = d; best = { kind: "h", idx: i }; }
});
return best;
}

// Use Pointer Events (covers mouse + touch + stylus on all modern browsers).
// Touch devices get a wider hit zone so fingers can grab thin lines.
cv.addEventListener("pointerdown", e => {
const isTouch = e.pointerType === "touch";
const hitPx = isTouch ? 20 : SplitInteract.HIT_PX;
const { x, y } = getCanvasCoord(e);
const hit = findHit(x, y, hitPx);
if (!hit) return;
e.preventDefault();
SplitInteract.drag = hit;
SplitInteract.hover = hit;
cv.setPointerCapture(e.pointerId);
drawSplitPreview();
});

cv.addEventListener("pointermove", e => {
if (SplitInteract.drag) {
const { x, y } = getCanvasCoord(e);
const drag = SplitInteract.drag;
const MIN_GAP = 0.02;
if (drag.kind === "v") {
const arr = AppState.split.vLines;
const lo = (drag.idx === 0 ? 0 : arr[drag.idx - 1]) + MIN_GAP;
const hi = (drag.idx === arr.length - 1 ? 1 : arr[drag.idx + 1]) - MIN_GAP;
arr[drag.idx] = clamp(x / cv.width, lo, hi);
} else {
const arr = AppState.split.hLines;
const lo = (drag.idx === 0 ? 0 : arr[drag.idx - 1]) + MIN_GAP;
const hi = (drag.idx === arr.length - 1 ? 1 : arr[drag.idx + 1]) - MIN_GAP;
arr[drag.idx] = clamp(y / cv.height, lo, hi);
}
drawSplitPreview();
return;
}
// Hover only meaningful for mouse devices
if (e.pointerType !== "touch") {
const { x, y } = getCanvasCoord(e);
const hit = findHit(x, y, SplitInteract.HIT_PX);
SplitInteract.hover = hit;
cv.classList.toggle("cursor-ew", hit && hit.kind === "v");
cv.classList.toggle("cursor-ns", hit && hit.kind === "h");
drawSplitPreview();
}
});

function endDrag(e) {
if (SplitInteract.drag) {
SplitInteract.drag = null;
try { cv.releasePointerCapture(e.pointerId); } catch (_) {}
drawSplitPreview();
}
}
cv.addEventListener("pointerup", endDrag);
cv.addEventListener("pointercancel", endDrag);

cv.addEventListener("pointerleave", e => {
if (!SplitInteract.drag && e.pointerType !== "touch") {
SplitInteract.hover = null;
cv.classList.remove("cursor-ew", "cursor-ns");
drawSplitPreview();
}
});
}

function runBackgroundRemoval() {
if (!AppState.split.frames.length) { toast("請先切割影格", "error"); return; }
if (!$("#bgRmEnable").checked) { toast("已停用智能去背，跳過", ""); return; }
AppState.split.frames.forEach(f => {
const ctx = f.canvas.getContext("2d", { willReadFrequently: true });
const id = ctx.getImageData(0, 0, f.canvas.width, f.canvas.height);
const d = id.data;
for (let i = 0; i < d.length; i += 4) {
const r = d[i], g = d[i+1], b = d[i+2];
// Green dominance check
const gDom = g - Math.max(r, b);
if (g > 100 && gDom > 30 && b < 200) {
// Soft mask: stronger gDom -> more transparent
let alpha = 1 - clamp((gDom - 20) / 80, 0, 1);
d[i+3] = Math.round(d[i+3] * alpha);
// De-spill: shift green channel toward avg of r,b
if (alpha < 0.95) {
const target = (r + b) / 2;
d[i+1] = Math.round(g * 0.4 + target * 0.6);
}
}
}
ctx.putImageData(id, 0, 0);
});
AppState.split.bgRemoved = true;
$("#bgRmStatusTag").textContent = "✅ 已完成去背";
$("#bgRmStatusTag").className = "tag success";
renderSplitFramesGrid();
toast("✅ 已完成 20 張影格的綠幕去背", "success");
}

/**

* After this, the editor will start fresh (no overlays) using the new image as bg.

f.dataURL = dataURL;

f.origDataURL = dataURL;

}

/* ----- Export final APNG / PNG ----- */
/* ----- APNG loop-count patching -----
* UPNG.js encodes APNG with num_plays = 0 (infinite) by default.
* LINE requires num_plays ∈ {1,2,3,4} AND total play time per loop ∈ {1,2,3,4} sec.
* Frame-duplication would make total time = totalSec × loops (e.g., 8s) → rejected.
* Correct approach: encode once at totalSec, then patch the acTL chunk's num_plays.
*/
const PNG_CRC_TABLE = (() => {
const t = new Uint32Array(256);
for (let n = 0; n < 256; n++) {
let c = n;
for (let k = 0; k < 8; k++) c = (c & 1) ? (0xEDB88320 ^ (c >>> 1)) : (c >>> 1);
t[n] = c >>> 0;
}
return t;
})();
function pngCrc32(bytes, start, len) {
let c = 0xFFFFFFFF;
for (let i = 0; i < len; i++) c = PNG_CRC_TABLE[(c ^ bytes[start + i]) & 0xFF] ^ (c >>> 8);
return (c ^ 0xFFFFFFFF) >>> 0;
}
function patchAPNGNumPlays(arrayBuffer, numPlays) {
const bytes = new Uint8Array(arrayBuffer);
const view = new DataView(bytes.buffer, bytes.byteOffset, bytes.byteLength);
let i = 8; // skip 8-byte PNG signature
while (i + 12 <= bytes.length) {
const len = view.getUint32(i);
const type = String.fromCharCode(bytes[i+4], bytes[i+5], bytes[i+6], bytes[i+7]);
if (type === "acTL") {
// acTL data layout: num_frames (uint32, big-endian) + num_plays (uint32, big-endian)
// num_plays sits at i+8 (data start) + 4 = i+12
view.setUint32(i + 12, numPlays >>> 0);
// Recompute CRC over chunk type + data (4 + len bytes starting at i+4)
const crc = pngCrc32(bytes, i + 4, 4 + len);
view.setUint32(i + 8 + len, crc);
return arrayBuffer;
}
i += 8 + len + 4; // length(4) + type(4) + data(len) + CRC(4)
}
return arrayBuffer; // no acTL found (e.g., static PNG) — return unchanged
}

/* Patch every fcTL chunk's delay_num/delay_den to express exact rational seconds.
* UPNG.js encodes delays as integer ms with delay_den=1000, which can have rounding errors:
* e.g. 1 sec / 6 frames = 166.67 ms each → UPNG rounds to 167 → total = 1002 ms ≠ 1.000 s.
* LINE rejects animations whose total duration isn't exactly an integer second.
* By writing the same fraction (num/den) to every fcTL, we guarantee N × (num/den) = num × N/den
* which we choose so the sum is exactly the requested number of seconds (mathematically perfect).
*
* fcTL chunk layout (26 bytes data):
* sequence_number (uint32) | width (uint32) | height (uint32) |
* x_offset (uint32) | y_offset (uint32) |
* delay_num (uint16) | delay_den (uint16) | ← we patch these
* dispose_op (uint8) | blend_op (uint8)
* delay_num at offset i+8+20 = i+28; delay_den at i+30.
*/
function patchAPNGFrameDelays(arrayBuffer, num, den) {
const bytes = new Uint8Array(arrayBuffer);
const view = new DataView(bytes.buffer, bytes.byteOffset, bytes.byteLength);
let i = 8;
let patched = 0;
while (i + 12 <= bytes.length) {
const len = view.getUint32(i);
const type = String.fromCharCode(bytes[i+4], bytes[i+5], bytes[i+6], bytes[i+7]);
if (type === "fcTL") {
view.setUint16(i + 28, num & 0xFFFF);
view.setUint16(i + 30, den & 0xFFFF);
const crc = pngCrc32(bytes, i + 4, 4 + len);
view.setUint32(i + 8 + len, crc);
patched++;
}
i += 8 + len + 4;
}
return arrayBuffer;
}

async function encodeForExport() {
// Builds the final blob (PNG for tab format, APNG for others) without triggering a download.
// Returns { blob, name } or throws on error.
const fmt = FormatSpec[AppState.compose.format];
const visible = AppState.compose.frames.filter(x => x.img && !x.hidden);
if (visible.length === 0) throw new Error("無可用影格");

if (AppState.compose.format === "tab") {
const c = document.createElement("canvas");
c.width = fmt.w; c.height = fmt.h;
const ctx = c.getContext("2d", { willReadFrequently: true });
drawFitImage(ctx, visible[0].img, 0, 0, fmt.w, fmt.h);
const blob = await canvasToBlob(c, "image/png");
return { blob, name: `tab_${fmt.w}x${fmt.h}_${Date.now()}.png` };
}

if (typeof UPNG === "undefined") throw new Error("APNG 編碼器尚未載入");
const totalSec = AppState.compose.totalSec; // 1, 2, 3, or 4
const N = visible.length; // frame count
const loops = AppState.compose.loop; // 1, 2, 3, or 4

const buffers = [];
const delaysMs = [];
const perMs = totalSec * 1000 / N; // approx ms (UPNG uses these initially)
for (let i = 0; i < N; i++) {
const c = document.createElement("canvas");
c.width = fmt.w; c.height = fmt.h;
const ctx = c.getContext("2d", { willReadFrequently: true });
drawFitImage(ctx, visible[i].img, 0, 0, fmt.w, fmt.h);
const id = ctx.getImageData(0, 0, fmt.w, fmt.h);
buffers.push(id.data.buffer);
delaysMs.push(Math.round(perMs)); // placeholder, will be overwritten by fcTL patch
}
await new Promise(r => setTimeout(r, 20)); // yield to UI

// Encode ONCE with the actual frame count + per-loop duration (NOT replicated).
let arr = UPNG.encode(buffers, fmt.w, fmt.h, 0, delaysMs);

// === Binary patches for mathematically exact LINE compliance ===
// 1. acTL.num_plays = loops (UPNG defaults to 0 = infinite loop, which LINE rejects)
arr = patchAPNGNumPlays(arr, loops);
// 2. fcTL.delay_num = totalSec, delay_den = N for every frame.
// Each frame's duration becomes EXACTLY totalSec / N seconds.
// Sum of N frames = N × (totalSec / N) = totalSec seconds (rational, no rounding).
arr = patchAPNGFrameDelays(arr, totalSec, N);

const blob = new Blob([arr], { type: "image/png" });
return { blob, name: `${AppState.compose.format}_${fmt.w}x${fmt.h}_${Date.now()}.png` };
}

/** Reflect current selectedObj's layer position in the side-panel UI.
* - infoSel: selector for the "x / N" label

const total = objects.length;
// Display layer position: 1 = bottom, total = top (so user thinks "higher number = on top")

const atBot = idx === 0;
grp.querySelectorAll("[data-act]").forEach(b => {
const a = b.dataset.act;
b.disabled = (atTop && (a === "front" || a === "forward")) ||

function changeZOrder(action) {
if (!selectedObj) return;
const i = objects.indexOf(selectedObj);
if (i < 0) return;
objects.splice(i, 1); // remove
let newIdx;
switch (action) {
case "front": newIdx = objects.length; break; // append (top)
case "back": newIdx = 0; break; // prepend (bottom)
case "forward": newIdx = Math.min(i + 1, objects.length); break;
case "backward": newIdx = Math.max(i - 1, 0); break;
default: newIdx = i;
}
objects.splice(newIdx, 0, selectedObj);
render();
// Refresh whichever panel is currently visible
if (selectedObj.type === "text") updateLayerInfo("#txtLayerInfo", "#txtLayerBtns");
else if (selectedObj.type === "deco") updateLayerInfo("#objLayerInfo", "#objLayerBtns");
}

function render() {
if (!ctx) return;
const fmt = FormatSpec[AppState.compose.format];
ctx.clearRect(0, 0, fmt.w, fmt.h);

// 1. background image
if (bgImg) {
const iw = bgImg.naturalWidth, ih = bgImg.naturalHeight;
const baseR = Math.min(fmt.w / iw, fmt.h / ih) * bgState.scale;
const dw = iw * baseR, dh = ih * baseR;
const dx = (fmt.w - dw) / 2 + bgState.x;
const dy = (fmt.h - dh) / 2 + bgState.y;
ctx.imageSmoothingQuality = "high";
ctx.drawImage(bgImg, dx, dy, dw, dh);
}

// 2. objects
objects.forEach(o => {
ctx.save();
ctx.translate(o.x, o.y);
ctx.rotate(o.rotation || 0);
ctx.scale(o.scale || 1, o.scale || 1);
if (o.type === "text") drawTextObject(o);
else if (o.type === "deco") drawDecoOnCtx(ctx, o);
ctx.restore();
});

// 3. selection
if (selectedObj) drawSelectionBox(selectedObj);
}

function drawTextObjectOnCtx(c, o) {
const prev = ctx; ctx = c;
drawTextObject(o);

}
function drawTextObject(o) {
const fontStr = `${o.size}px ${o.font}`;
ctx.font = fontStr;
ctx.textAlign = "center"; ctx.textBaseline = "middle";
ctx.lineJoin = "round";
if (o.direction === "h") {
if (o.strokeW > 0) {
ctx.lineWidth = o.strokeW; ctx.strokeStyle = o.stroke;
ctx.strokeText(o.text, 0, 0);
}
ctx.fillStyle = o.color;
ctx.fillText(o.text, 0, 0);
} else {
// vertical text
const chars = Array.from(o.text);
const total = chars.length;
chars.forEach((ch, i) => {
const yy = (i - (total - 1) / 2) * o.size;
if (o.strokeW > 0) {
ctx.lineWidth = o.strokeW; ctx.strokeStyle = o.stroke;
ctx.strokeText(ch, 0, yy);
}
ctx.fillStyle = o.color;
ctx.fillText(ch, 0, yy);
});
}
}

function getTextBounds(o) {
ctx.save();
ctx.font = `${o.size}px ${o.font}`;
if (o.direction === "h") {
const m = ctx.measureText(o.text);
const w = (m.width + o.strokeW) * (o.scale || 1);
const h = (o.size + o.strokeW) * (o.scale || 1);
ctx.restore();
return { w, h };
} else {
const chars = Array.from(o.text);
let maxW = 0;
chars.forEach(c => { maxW = Math.max(maxW, ctx.measureText(c).width); });
const w = (maxW + o.strokeW) * (o.scale || 1);
const h = (chars.length * o.size + o.strokeW) * (o.scale || 1);
ctx.restore();
return { w, h };
}
}

function getObjectBounds(o) {
if (o.type === "text") return getTextBounds(o);
// deco: stored bounding size = 80 base, scale outward
const base = 80;
return { w: base * o.scale, h: base * o.scale };
}

const armLen = 4;
ctx.beginPath();
ctx.moveTo(xX - armLen, xY - armLen); ctx.lineTo(xX + armLen, xY + armLen);
ctx.moveTo(xX - armLen, xY + armLen); ctx.lineTo(xX + armLen, xY - armLen);
ctx.stroke();
// Rotation handle: line + circle at top-center, above the box
const rotR = 9;
const rotOffset = 24; // px above the bounding box top edge
const rotY = -b.h/2 - rotOffset;
// Connection line from top-center to handle
ctx.strokeStyle = "#06C755"; ctx.lineWidth = 1.2;
ctx.setLineDash([3, 2]);
ctx.beginPath();
ctx.moveTo(0, -b.h/2); ctx.lineTo(0, rotY + rotR);
ctx.stroke();
ctx.setLineDash([]);
// Handle circle (LINE green to differentiate from delete/resize)
ctx.fillStyle = "#06C755";
ctx.beginPath(); ctx.arc(0, rotY, rotR, 0, Math.PI * 2); ctx.fill();
ctx.strokeStyle = "#fff"; ctx.lineWidth = 2; ctx.stroke();
// Curved arrow icon inside the handle (270° arc + arrowhead)
ctx.strokeStyle = "#fff"; ctx.lineWidth = 1.6;
const arcR = 4.2;
ctx.beginPath();
ctx.arc(0, rotY, arcR, Math.PI * 0.25, Math.PI * 1.75);
ctx.stroke();
// Arrowhead at start of arc
ctx.fillStyle = "#fff";
const ax = Math.cos(Math.PI * 0.25) * arcR;
const ay = rotY + Math.sin(Math.PI * 0.25) * arcR;
ctx.beginPath();
ctx.moveTo(ax + 2, ay - 1);
ctx.lineTo(ax - 1, ay - 2);
ctx.lineTo(ax + 1, ay + 2.5);
ctx.closePath();
ctx.fill();
ctx.restore();
}

function getCanvasCoords(e) {
const rect = cv.getBoundingClientRect();
const x = (e.clientX - rect.left) * cv.width / rect.width;
const y = (e.clientY - rect.top) * cv.height / rect.height;
return { x, y };
}

// Convert world coordinates → object's local (un-rotated) space.
// After this, hit-testing can simply compare to the un-rotated bounding box.
function worldToLocal(o, wx, wy) {
const dx = wx - o.x, dy = wy - o.y;
const r = -(o.rotation || 0);
const c = Math.cos(r), s = Math.sin(r);
return { lx: dx * c - dy * s, ly: dx * s + dy * c };
}

function findObjectAt(x, y) {
for (let i = objects.length - 1; i >= 0; i--) {
const o = objects[i];
const b = getObjectBounds(o);
const { lx, ly } = worldToLocal(o, x, y);
if (lx >= -b.w/2 && lx <= b.w/2 && ly >= -b.h/2 && ly <= b.h/2) return o;
}
return null;
}
function isInResizeHandle(o, x, y) {
const b = getObjectBounds(o);
const { lx, ly } = worldToLocal(o, x, y);
return Math.abs(lx - b.w/2) < 10 && Math.abs(ly - b.h/2) < 10;
}
function isInDeleteButton(o, x, y) {
const b = getObjectBounds(o);
const { lx, ly } = worldToLocal(o, x, y);
return Math.hypot(lx + b.w/2, ly + b.h/2) < 12;
}
function isInRotateHandle(o, x, y) {
const b = getObjectBounds(o);
const { lx, ly } = worldToLocal(o, x, y);
return Math.hypot(lx, ly + b.h/2 + 24) < 12; // 24 = rotOffset in drawSelectionBox
}

function onMouseDown(e) {
const { x, y } = getCanvasCoords(e);
if (pickingFor) return;
e.preventDefault();
// Check the X delete button FIRST (top priority over resize/move)
if (selectedObj && isInDeleteButton(selectedObj, x, y)) {
objects = objects.filter(o => o !== selectedObj);
selectedObj = null;
$("#textProps").style.display = "none";
$("#objStyleSection").style.display = "none";
render();
toast("已刪除物件", "");
return;
}
if (selectedObj && isInRotateHandle(selectedObj, x, y)) {
drag = {
kind: "rotate",
obj: selectedObj,
startMouseAng: Math.atan2(y - selectedObj.y, x - selectedObj.x),
startObjAng: selectedObj.rotation || 0
};
return;
}
if (selectedObj && isInResizeHandle(selectedObj, x, y)) {
drag = { kind: "resize", startScale: selectedObj.scale, startDist: Math.hypot(x - selectedObj.x, y - selectedObj.y), obj: selectedObj };
return;
}
const o = findObjectAt(x, y);
if (o) {
selectedObj = o;
drag = { kind: "move", offX: x - o.x, offY: y - o.y, obj: o };
onSelect(o);
render();
} else {
selectedObj = null;
// bg drag
drag = { kind: "bg", startX: x, startY: y, startBgX: bgState.x, startBgY: bgState.y };
$("#textProps").style.display = "none";
$("#objStyleSection").style.display = "none";
render();
}
}
function onMouseMove(e) {
if (!drag) return;
const { x, y } = getCanvasCoords(e);
if (drag.kind === "move") {
drag.obj.x = x - drag.offX;
drag.obj.y = y - drag.offY;
render();
} else if (drag.kind === "resize") {
const dist = Math.hypot(x - drag.obj.x, y - drag.obj.y);
drag.obj.scale = clamp(drag.startScale * (dist / Math.max(drag.startDist, 1)), 0.1, 8);
render();
} else if (drag.kind === "rotate") {
const ang = Math.atan2(y - drag.obj.y, x - drag.obj.x);
let newRot = drag.startObjAng + (ang - drag.startMouseAng);
// Hold Shift for 15° snap (typical design-tool convention)
if (e.shiftKey) {
const step = Math.PI / 12; // 15 degrees
newRot = Math.round(newRot / step) * step;
}
drag.obj.rotation = newRot;
// Sync the side-panel slider so the user sees angle update live
const deg = Math.round(newRot * 180 / Math.PI);
const wrappedDeg = ((deg % 360) + 540) % 360 - 180;
if (drag.obj.type === "text") {
if ($("#txtRot")) { $("#txtRot").value = wrappedDeg; $("#txtRotVal").textContent = deg + "°"; }
} else if (drag.obj.type === "deco") {
if ($("#objRot")) { $("#objRot").value = wrappedDeg; $("#objRotVal").textContent = deg + "°"; }
}
render();
} else if (drag.kind === "bg") {
bgState.x = drag.startBgX + (x - drag.startX);
bgState.y = drag.startBgY + (y - drag.startY);
render();
}
}
function onMouseUp() { drag = null; }

function onSelect(o) {
if (o.type === "text") {
$("#textProps").style.display = "block";
$("#objStyleSection").style.display = "none";
reflectTextProps();
setTimeout(() => $("#textProps").scrollIntoView({ behavior: "smooth", block: "start" }), 50);
} else if (o.type === "deco") {
$("#objStyleSection").style.display = "block";
reflectObjProps();
setTimeout(() => $("#objStyleSection").scrollIntoView({ behavior: "smooth", block: "start" }), 50);
}
}

function onCanvasHover(e) {
if (!pickingFor) {
if (selectedObj) {
const { x, y } = getCanvasCoords(e);
if (isInDeleteButton(selectedObj, x, y)) cv.style.cursor = "pointer";
else if (isInRotateHandle(selectedObj, x, y)) cv.style.cursor = "grab";
else if (isInResizeHandle(selectedObj, x, y)) cv.style.cursor = "nwse-resize";
else cv.style.cursor = "move";
} else { cv.style.cursor = ""; }
}
}

function onCanvasClick(e) {
if (!pickingFor) return;
const { x, y } = getCanvasCoords(e);
const id = ctx.getImageData(Math.round(x), Math.round(y), 1, 1).data;
const hex = "#" + [id[0], id[1], id[2]].map(v => v.toString(16).padStart(2, "0")).join("");
if (pickingFor === "txtColor" && selectedObj?.type === "text") { selectedObj.color = hex; $("#txtColor").value = hex; $("#txtColorVal").textContent = hex; }
if (pickingFor === "txtStroke" && selectedObj?.type === "text") { selectedObj.stroke = hex; $("#txtStroke").value = hex; $("#txtStrokeVal").textContent = hex; }
if (pickingFor === "objFill" && selectedObj?.type === "deco") { selectedObj.fill = hex; selectedObj.noFill = false; $("#objFill").value = hex; $("#objFillNone").checked = false; }
if (pickingFor === "objStroke" && selectedObj?.type === "deco") { selectedObj.stroke = hex; $("#objStroke").value = hex; }
stopPicker();
render();
}

function saveAndClose() {
// Re-render to a fresh canvas without selection, then store back
const fmt = FormatSpec[AppState.compose.format];
const out = document.createElement("canvas");
out.width = fmt.w; out.height = fmt.h;
const octx = out.getContext("2d", { willReadFrequently: true });

// bg
if (bgImg) {
const iw = bgImg.naturalWidth, ih = bgImg.naturalHeight;
const baseR = Math.min(fmt.w / iw, fmt.h / ih) * bgState.scale;
const dw = iw * baseR, dh = ih * baseR;
const dx = (fmt.w - dw) / 2 + bgState.x;
const dy = (fmt.h - dh) / 2 + bgState.y;
octx.imageSmoothingQuality = "high";
octx.drawImage(bgImg, dx, dy, dw, dh);
}
// objects
objects.forEach(o => {
octx.save();
octx.translate(o.x, o.y);
octx.rotate(o.rotation || 0);
octx.scale(o.scale || 1, o.scale || 1);
if (o.type === "text") drawTextObjectOnCtx(octx, o);
else if (o.type === "deco") drawDecoOnCtx(octx, o);
octx.restore();
});

const url = out.toDataURL("image/png");
dataURLToImage(url).then(img => {
const frame = AppState.compose.frames[frameIdx];
// Update the displayed/exported (baked) image
frame.img = img;
frame.dataURL = url;
// Persist editor state so user can re-open and continue editing
// (origImg stays untouched as the pristine bg for future edits)
frame.editorState = {
bgState: { ...bgState },
objects: objects.map(o => ({ ...o }))
};
composeRefreshAll();
toast("✅ 已儲存編輯（可再次開啟編輯器修改）", "success");
close();
});
}

return { open };
})();
