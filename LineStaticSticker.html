// 用 textarea + execCommand 實作複製（避開 iframe 中 navigator.clipboard 權限問題）
function copyText(text) {
const ta = document.createElement("textarea");
ta.value = text;
ta.style.position = "fixed";
ta.style.left = "-9999px";
ta.style.top = "0";
document.body.appendChild(ta);
ta.focus();
ta.select();
let ok = false;
try { ok = document.execCommand("copy"); } catch(e) { ok = false; }
document.body.removeChild(ta);
return ok;
}

function toggleScenario(chip, txt) {

$$("#scenarios-wrap .chip").forEach(c => {

c.classList.toggle("disabled", PromptState.scenarios.size >= 20);

c.classList.remove("disabled");
}
});
updatePrompt();
}

// 雙線性插值取 alpha
function sampleAlpha(arr, x, y) {
if (x < 0 || y < 0 || x > 99 || y > 99) return 0;
const x0 = Math.floor(x), y0 = Math.floor(y);
const x1 = Math.min(x0 + 1, 99), y1 = Math.min(y0 + 1, 99);
const dx = x - x0, dy = y - y0;
const a00 = arr[y0 * 100 + x0];
const a10 = arr[y0 * 100 + x1];
const a01 = arr[y1 * 100 + x0];
const a11 = arr[y1 * 100 + x1];
return a00 * (1-dx)*(1-dy) + a10 * dx*(1-dy) + a01 * (1-dx)*dy + a11 * dx*dy;
}

function processWatermark() {
// 從原圖複製到輸出
const out = new ImageData(
new Uint8ClampedArray(WM.origImage.data),
WM.width, WM.height
);
// 開關關閉時直接回傳原圖（不做任何處理）
if (!WM.enabled) return out;
if (!WM.alphaArr) return out;

const tplW = 100 * WM.scale;
const tplH = 100 * WM.scale;
const x0 = Math.round(WM.width - WM.marginR - tplW);
const y0 = Math.round(WM.height - WM.marginB - tplH);

const xStart = Math.max(0, x0);
const yStart = Math.max(0, y0);
const xEnd = Math.min(WM.width, x0 + Math.ceil(tplW));
const yEnd = Math.min(WM.height, y0 + Math.ceil(tplH));

const data = out.data;
const orig = WM.origImage.data;

for (let py = yStart; py < yEnd; py++) {
for (let px = xStart; px < xEnd; px++) {
// 對應到模板的浮點座標 (0~99)
const tx = (px - x0) / WM.scale;
const ty = (py - y0) / WM.scale;
let a = sampleAlpha(WM.alphaArr, tx, ty) * WM.alphaMul;
if (a <= 0) continue;
if (a > 0.985) a = 0.985;
const inv = 1 - a;
const idx = (py * WM.width + px) * 4;
// bg = (obs - 255*a) / (1 - a)
const r = (orig[idx] - 255 * a) / inv;
const g = (orig[idx+1] - 255 * a) / inv;
const b = (orig[idx+2] - 255 * a) / inv;
data[idx] = r < 0 ? 0 : r > 255 ? 255 : r;
data[idx+1] = g < 0 ? 0 : g > 255 ? 255 : g;
data[idx+2] = b < 0 ? 0 : b > 255 ? 255 : b;
}
}
return out;
}

function renderWm() {

if (WM.view === "original") {
ctx.putImageData(WM.origImage, 0, 0);
} else {
const out = processWatermark();
ctx.putImageData(out, 0, 0);
WM._lastProcessed = out;
}

// 紅色虛線定位框

const tplW = 100 * WM.scale;
const tplH = 100 * WM.scale;
const x0 = WM.width - WM.marginR - tplW;
const y0 = WM.height - WM.marginB - tplH;
ctx.save();
// 線寬隨圖片尺寸縮放，避免顯示太細
const lw = Math.max(2, Math.round(WM.width / 600));
ctx.strokeStyle = "#ff3344";
ctx.lineWidth = lw;
ctx.setLineDash([lw * 3, lw * 2]);
ctx.strokeRect(x0, y0, tplW, tplH);
ctx.restore();
}
$("wm-area").textContent = `${Math.round(100*WM.scale)}×${Math.round(100*WM.scale)} @ R${WM.marginR}/B${WM.marginB}`;
}

// 綁定 canvas 上的分割線拖曳與 hover 行為（鼠標 + 觸控）
function bindSplitCanvasInteraction() {
const canvas = $("split-canvas");

// 螢幕座標 → canvas 內部座標
function getCanvasCoords(e) {
const rect = canvas.getBoundingClientRect();
const cx = (e.touches && e.touches[0]) ? e.touches[0].clientX
: (e.changedTouches && e.changedTouches[0]) ? e.changedTouches[0].clientX
: e.clientX;
const cy = (e.touches && e.touches[0]) ? e.touches[0].clientY
: (e.changedTouches && e.changedTouches[0]) ? e.changedTouches[0].clientY
: e.clientY;
return {
x: (cx - rect.left) * canvas.width / rect.width,
y: (cy - rect.top) * canvas.height / rect.height,
scaleX: canvas.width / rect.width,
scaleY: canvas.height / rect.height
};
}

// 點擊命中測試（傳回最近的線；同時命中時優先垂直）
function hitTest(c) {
if (!Split.colLines.length) return null;
const W = Split.fullW, H = Split.fullH;
// 顯示上 ~16px 的命中半徑
const tx = 16 * c.scaleX, ty = 16 * c.scaleY;
let best = null, bestD = Infinity;
Split.colLines.forEach((n, i) => {
const d = Math.abs(c.x - n*W);
if (d < tx && d < bestD) { best = { type: "col", index: i }; bestD = d; }
});
Split.rowLines.forEach((n, j) => {
const d = Math.abs(c.y - n*H);
if (d < ty && d < bestD) { best = { type: "row", index: j }; bestD = d; }
});
return best;
}

function applyDrag(c) {
if (!Split.dragLine) return;
const W = Split.fullW, H = Split.fullH;
const minGap = 0.04; // 相鄰兩線最小距離（佔 4%）
if (Split.dragLine.type === "col") {
const i = Split.dragLine.index, arr = Split.colLines;
const prev = i > 0 ? arr[i-1] : 0;
const next = i < arr.length - 1 ? arr[i+1] : 1;
arr[i] = Math.max(prev + minGap, Math.min(next - minGap, c.x / W));
} else {
const j = Split.dragLine.index, arr = Split.rowLines;
const prev = j > 0 ? arr[j-1] : 0;
const next = j < arr.length - 1 ? arr[j+1] : 1;
arr[j] = Math.max(prev + minGap, Math.min(next - minGap, c.y / H));
}
}

function onDown(e) {
if (!Split.sourceCanvas) return;
const c = getCanvasCoords(e);
const h = hitTest(c);
if (h) {
e.preventDefault();
Split.dragLine = h;
// 立刻吸附到指針
applyDrag(c);
renderSplitPreview();
}
}

function onMove(e) {
if (!Split.sourceCanvas) return;
const c = getCanvasCoords(e);
if (Split.dragLine) {
e.preventDefault();
applyDrag(c);
renderSplitPreview();
} else {
const h = hitTest(c);
const newCursor = h ? (h.type === "col" ? "col-resize" : "row-resize") : "";
if (canvas.style.cursor !== newCursor) canvas.style.cursor = newCursor;
const a = JSON.stringify(h), b = JSON.stringify(Split.hoverLine);
if (a !== b) { Split.hoverLine = h; renderSplitPreview(); }
}
}

function onUp() {
if (Split.dragLine) {
Split.dragLine = null;
renderSplitPreview();
}
}

canvas.addEventListener("mousedown", onDown);
window.addEventListener("mousemove", onMove);
window.addEventListener("mouseup", onUp);
canvas.addEventListener("touchstart", onDown, { passive: false });
canvas.addEventListener("touchmove", onMove, { passive: false });
canvas.addEventListener("touchend", onUp);
canvas.addEventListener("touchcancel", onUp);
canvas.addEventListener("mouseleave", () => {
if (!Split.dragLine && Split.hoverLine) {
Split.hoverLine = null;
canvas.style.cursor = "";
renderSplitPreview();
}
});
}

function renderSplitPreview() {
if (!Split.sourceCanvas) return;
const canvas = $("split-canvas");
const ctx = canvas.getContext("2d", { willReadFrequently: true });
const W = Split.fullW, H = Split.fullH;
canvas.width = W; canvas.height = H;
ctx.fillStyle = "#000";
ctx.fillRect(0, 0, W, H);
// 畫帶 offset/zoom 的圖片
const dw = W * Split.zoom;
const dh = H * Split.zoom;
const dx = (W - dw)/2 + Split.offsetX;
const dy = (H - dh)/2 + Split.offsetY;
ctx.drawImage(Split.sourceCanvas, dx, dy, dw, dh);

// 確保分割線已初始化
if (!Split.colLines.length || !Split.rowLines.length) resetSplitLines();

// 畫可拖曳分割線（含中央把手）
const lw = Math.max(2, W/600);
const handleR = lw * 6;
ctx.save();

const drawLine = (isVertical, idx, n) => {
const isActive = Split.dragLine && Split.dragLine.type === (isVertical?"col":"row") && Split.dragLine.index === idx;
const isHover = Split.hoverLine && Split.hoverLine.type === (isVertical?"col":"row") && Split.hoverLine.index === idx;
const color = isActive ? "#ffffff" : (isHover ? "#fbbf24" : "#f95f8e");

// 線
ctx.strokeStyle = color;
ctx.lineWidth = isActive || isHover ? lw*1.6 : lw;
ctx.setLineDash([lw*4, lw*3]);
ctx.beginPath();
if (isVertical) { ctx.moveTo(n*W, 0); ctx.lineTo(n*W, H); }
else { ctx.moveTo(0, n*H); ctx.lineTo(W, n*H); }
ctx.stroke();

// 把手（兩個端點 + 中心，提升點擊命中率與可發現性）
ctx.setLineDash([]);
ctx.fillStyle = color;
ctx.strokeStyle = "rgba(0,0,0,.4)";
ctx.lineWidth = lw*0.6;
const hr = isActive || isHover ? handleR*1.3 : handleR;
const handlePts = isVertical
? [[n*W, H*0.15], [n*W, H/2], [n*W, H*0.85]]
: [[W*0.15, n*H], [W/2, n*H], [W*0.85, n*H]];
handlePts.forEach(([px, py]) => {
ctx.beginPath(); ctx.arc(px, py, hr, 0, Math.PI*2);
ctx.fill(); ctx.stroke();
});
};

Split.colLines.forEach((n, i) => drawLine(true, i, n));
Split.rowLines.forEach((n, j) => drawLine(false, j, n));
ctx.restore();

// 同步更新 HTML overlay 上的紅色 × 刪除按鈕位置
renderSplitOverlay();
}

/* =========================================================================
* 純 JS ZIP 建構器（STORED / 不壓縮）
* PNG 已是壓縮格式，再壓收益極低，stored 即可
* ======================================================================= */
let _crcTable = null;
function makeCrcTable() {
_crcTable = new Uint32Array(256);
for (let n = 0; n < 256; n++) {
let c = n;
for (let k = 0; k < 8; k++) c = (c & 1) ? (0xEDB88320 ^ (c >>> 1)) : (c >>> 1);
_crcTable[n] = c;
}
}
function crc32(data) {
if (!_crcTable) makeCrcTable();
let crc = 0xFFFFFFFF;
for (let i = 0; i < data.length; i++) {
crc = _crcTable[(crc ^ data[i]) & 0xFF] ^ (crc >>> 8);
}
return (crc ^ 0xFFFFFFFF) >>> 0;
}
function strToBytes(s) {
return new TextEncoder().encode(s);
}
function writeU16(arr, off, v) { arr[off] = v & 0xFF; arr[off+1] = (v >>> 8) & 0xFF; }
function writeU32(arr, off, v) { arr[off] = v & 0xFF; arr[off+1] = (v >>> 8) & 0xFF; arr[off+2] = (v >>> 16) & 0xFF; arr[off+3] = (v >>> 24) & 0xFF; }

function createZip(files) {
const localParts = [];
const centralParts = [];
let offset = 0;

files.forEach(f => {
const nameBytes = strToBytes(f.name);
const data = f.data;
const crc = crc32(data);
const size = data.length;

// Local file header (30 bytes + name)
const lh = new Uint8Array(30 + nameBytes.length);
writeU32(lh, 0, 0x04034b50); // signature
writeU16(lh, 4, 20); // version needed
writeU16(lh, 6, 0); // flags
writeU16(lh, 8, 0); // method = stored
writeU16(lh, 10, 0); // mod time
writeU16(lh, 12, 0); // mod date
writeU32(lh, 14, crc); // crc32
writeU32(lh, 18, size); // compressed size
writeU32(lh, 22, size); // uncompressed size
writeU16(lh, 26, nameBytes.length);
writeU16(lh, 28, 0); // extra length
lh.set(nameBytes, 30);
localParts.push(lh, data);

// Central directory entry (46 bytes + name)
const cd = new Uint8Array(46 + nameBytes.length);
writeU32(cd, 0, 0x02014b50); // signature
writeU16(cd, 4, 20); // version made by
writeU16(cd, 6, 20); // version needed
writeU16(cd, 8, 0); // flags
writeU16(cd, 10, 0); // method
writeU16(cd, 12, 0); // mod time
writeU16(cd, 14, 0); // mod date
writeU32(cd, 16, crc);
writeU32(cd, 20, size);
writeU32(cd, 24, size);
writeU16(cd, 28, nameBytes.length);
writeU16(cd, 30, 0); // extra
writeU16(cd, 32, 0); // comment
writeU16(cd, 34, 0); // disk number
writeU16(cd, 36, 0); // internal attrs
writeU32(cd, 38, 0); // external attrs
writeU32(cd, 42, offset); // local header offset
cd.set(nameBytes, 46);
centralParts.push(cd);

offset += lh.length + data.length;
});

// 計算 central directory 區段大小
const cdSize = centralParts.reduce((a, b) => a + b.length, 0);
const cdOffset = offset;

// End of central directory record (22 bytes)
const eocd = new Uint8Array(22);
writeU32(eocd, 0, 0x06054b50);
writeU16(eocd, 4, 0); // disk
writeU16(eocd, 6, 0); // disk with cd
writeU16(eocd, 8, files.length); // entries on this disk
writeU16(eocd, 10, files.length); // total entries
writeU32(eocd, 12, cdSize);
writeU32(eocd, 16, cdOffset);
writeU16(eocd, 20, 0); // comment length

// 合併
const totalSize = offset + cdSize + 22;
const out = new Uint8Array(totalSize);
let p = 0;
localParts.forEach(part => { out.set(part, p); p += part.length; });
centralParts.forEach(part => { out.set(part, p); p += part.length; });
out.set(eocd, p);
return out;
}
